\begin{lemma}
Nếu $I$ là tâm đường tròn nội tiếp $\triangle ABC$ ta có
\[
	\angle BIC = 90 \degree + \frac{1}{2}A
\]
\end{lemma}

\begin{center}
\includesvg{l2}
\end{center}

\begin{proof}
	Ta có 
	\begin{align*}
		\angle BIC &= 180 \degree - (\angle IBC + \angle ICB) \\ 
				   &= 180 \degree - \frac{1}{2}(B + C) \\ 
				   &= 180 \degree - \frac{1}{2}(180 \degree - A) \\
				   &= 90 \degree + \frac{1}{2}A
	\end{align*}
\end{proof}
